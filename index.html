Timing execution of code with .time 
            
function wait(ms){
    var start = new Date().getTime();
    var end = start;
    while(end < start + ms) {
        end = new Date().getTime();
    }
}

console.log("Start long running process...")
console.time("process");
wait(3000);
console.log("Log info about the timer...")
console.timeLog("process");
wait(3000);
console.log("End the timer...")
console.timeEnd("process");
Group logging with .group command 
            
for(i=1;i<=3;i++) {
    console.count("i loop");
    console.group("J counter logs");
    for(j=1;j<=3;j++) {
        console.count("==> j loop");
    }
    console.groupEnd("J counter logs");
}
