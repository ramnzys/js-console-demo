Obteniendo el tiempo de ejecución con .time 
            
function wait(ms){
    var start = new Date().getTime();
    var end = start;
    while(end < start + ms) {
        end = new Date().getTime();
    }
}

console.log("Inicio de un proceso largo...")
console.time("process");
wait(3000);
console.log("Avance de un proceso largo...")
console.timeLog("process");
wait(3000);
console.log("Fin del proceso...")
console.timeEnd("process");
Mensajes agrupados con comando .group y .groupCollapsed 
            
for(i=1;i<=3;i++) {
    console.count("Contador i");
    console.group("Mensaje agrupados");
    for(j=1;j<=3;j++) {
        console.count("==> Contador j");
    }
    console.groupEnd("Mensajes agrupados");
}

for(i=1;i<=3;i++) {
    console.count("Contador i");
    console.groupCollapsed("Grupo colapsado");
    for(j=1;j<=3;j++) {
        console.count("==> Contador j");
    }
    console.groupEnd("Grupo colapsado");
}
